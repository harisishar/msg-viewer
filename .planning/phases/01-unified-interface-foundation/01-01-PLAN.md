```typescript
export interface UnifiedMessage {
  content: UnifiedMessageContent;
  attachments: UnifiedAttachment[];
  recipients: UnifiedRecipient[];
}

export interface UnifiedMessageContent {
  date: Date | null;
  subject: string;
  senderName: string;
  senderEmail: string;
  body: string;
  bodyHTML: string;
  bodyRTF?: DataView;
  headers: string;
}

export interface UnifiedAttachment {
  fileName: string;
  displayName: string;
  mimeType: string;
  content?: Uint8Array;
  embeddedMessage?: UnifiedMessage;
}

export interface UnifiedRecipient {
  name: string;
  email: string;
  type: 'to' | 'cc' | 'bcc';
}
```

Key decisions:
- `bodyRTF` is typed as `DataView` (not `Uint8Array`) because the RTF decompressor (`decompressRTF(view: DataView)`) consumes DataView directly — no unnecessary conversion.
- `file: CompoundFile` is DROPPED — OLE2 internal, must not appear in unified interface.
- `toRecipients`/`ccRecipients` strings are DROPPED — replaced by `type` field on `UnifiedRecipient`.
- `extension` and `language` fields from old Attachment are DROPPED — not used by any UI component and not available in EML format.
- `content` is `Uint8Array` (browser-native) instead of `Buffer` (Node.js type annotation that doesn't match runtime reality).
- `embeddedMsgObj: DirectoryEntry` replaced by `embeddedMessage?: UnifiedMessage` (pre-parsed, not lazy).

   ```typescript
   function mapAttachments(file: CompoundFile, rawAttachments: RawAttachment[]): UnifiedAttachment[] {
     return rawAttachments.map(raw => {
       const attachment: UnifiedAttachment = {
         fileName: raw.fileName ?? "",
         displayName: raw.displayName ?? "",
         mimeType: raw.mimeType ?? "",
       };

       if (raw.embeddedMsgObj) {
         // Property 3701 as PtypObject — embedded message
         attachment.embeddedMessage = parseDir(file, raw.embeddedMsgObj as unknown as DirectoryEntry);
       } else if (raw.content) {
         // Property 3701 as PtypBinary — binary attachment
         const dv = raw.content as DataView;
         attachment.content = new Uint8Array(dv.buffer, dv.byteOffset, dv.byteLength);
       }

       return attachment;
     });
   }
   ```

7. **Add `mapRecipients()` function** — port the null-byte-aware classification from `recipient/index.ts`'s `toSet()`:

   ```typescript
   function mapRecipients(rawContent: RawContent, rawRecipients: RawRecipient[]): UnifiedRecipient[] {
     const toSet = parseRecipientString(rawContent.toRecipients);
     const ccSet = parseRecipientString(rawContent.ccRecipients);

     return rawRecipients.map(raw => ({
       name: raw.name ?? "",
       email: raw.email ?? "",
       type: toSet.has(raw.name) ? 'to' as const
           : ccSet.has(raw.name) ? 'cc' as const
           : 'to' as const,  // default to 'to' if not found
     }));
   }

   function parseRecipientString(recipStr: string): Set<string> {
     if (!recipStr) return new Set();
     return new Set(
       recipStr.endsWith('\x00')
         ? recipStr.slice(0, -1).split("; ")
         : recipStr.split("; ")
     );
   }
   ```
